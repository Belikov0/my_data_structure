bool isPalindrome(char* A){
    int end =  strlen(A)-1;
    int start = 0;
    int flag = 1;
    while (start < end)
    {
        if(A[start++] != A[end--])
        {
            flag = 0;
            break;
        }

    }
    return flag;
}
